Il modulo va "attaccato" all'Html tramite la direttiva angular ng-app.

 <div ng-app="myApp">
  <div ng-controller="myController">
      { { variable } }
  </div>



Le componenti di angular vanno definiti dal modulo. È possibile definire una variabile per referenziare il modulo.
    
    var myAppModule = angular.module('myApp', [
        //array delle dipendenze da altri moduli
    ]);

    myAppModule.controller('myController', function($scope) {
        $scope.variable = "Hello World!";
    });
