Aggiungiamo questa direttiva personalizzata

var myModule.directive('digitalClock', [
    '$interval',
    '$filter',
    function ($interval, $filter) {

        return {
            restrict: 'EA',
            template: '-> { { time } } <-',
            replace: false,
            scope: true,
            link: function (scope, element, attrs) {

                scope.format = attrs.format || 'HH:mm:ss';
                
                $interval(function(){
                      scope.time = $filter('date')(new Date(), scope.format);
                }, 1000);
                // ogni secondo esegue la funzione
            }
        };
        
    }]);
